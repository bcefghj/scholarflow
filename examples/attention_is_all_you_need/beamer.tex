\begin{frame}{Attention Is All You Need}


\begin{itemize}

    \item Transformer: 完全基于注意力机制的新型网络架构

    \item Ashish Vaswani, Noam Shazeer, Niki Parmar 等 Google 团队

    \item NeurIPS 2017 (2017年12月)

    \item 首次提出摒弃RNN和卷积，仅使用注意力机制进行序列建模

    \item 开创了深度学习自然语言处理的新时代

\end{itemize}


\end{frame}


\begin{frame}{研究背景与动机}


\begin{itemize}

    \item RNN、LSTM和GRU是序列建模的主流方法

    \item RNN的顺序计算特性限制了并行化能力

    \item 序列越长，内存约束越严重，批处理困难

    \item 注意力机制可建模任意距离的依赖关系

    \item 现有模型将注意力与RNN结合，非纯注意力架构

\end{itemize}


\end{frame}


\begin{frame}{相关工作}


\begin{itemize}

    \item ByteNet和ConvS2S使用卷积实现并行计算

    \item 但连接任意位置的操作数随距离线性或对数增长

    \item Self-attention已用于阅读理解、摘要等任务

    \item 端到端记忆网络基于注意力而非时序循环

    \item Transformer是首个完全基于自注意力的模型

\end{itemize}


\end{frame}


\begin{frame}{Transformer 架构概览}

\begin{columns}[T]
\begin{column}{0.55\textwidth}

\begin{itemize}

    \item 采用标准的Encoder-Decoder结构

    \item 完全基于注意力机制，摒弃循环和卷积

    \item 堆叠6层相同的Encoder和Decoder模块

    \item 每个子层使用残差连接和层归一化

    \item 所有子层输出维度 dmodel = 512

\end{itemize}

\begin{frame}{注意力机制：Scaled Dot-Product Attention}

\begin{columns}[T]
\begin{column}{0.55\textwidth}

\begin{itemize}

    \item 输入：Query Q, Key K, Value V 三个向量

    \item 计算Q与所有K的点积，除以√dk进行缩放

    \item 应用softmax获得注意力权重

    \item 公式：Attention(Q,K,V) = softmax(QKT/√dk)V

    \item 缩放因子防止点积过大导致梯度消失

\end{itemize}

\end{column}

\begin{frame}{多头注意力机制 (Multi-Head Attention)}

\begin{columns}[T]
\begin{column}{0.55\textwidth}

\begin{itemize}

    \item 将Q、K、V投影到h个不同的子空间（h=8）

    \item 每个头维度 dk = dv = dmodel/h = 64

    \item 并行计算各头的注意力后拼接

    \item 公式：MultiHead(Q,K,V) = Concat(head1,...,headh)WO

    \item 多头机制允许同时关注不同表示子空间的信息

\end{itemize}

\end{column}

\begin{frame}{Transformer 中的三种注意力应用}

\begin{columns}[T]
\begin{column}{0.55\textwidth}

\begin{itemize}

    \item Encoder Self-Attention: 所有K、V、Q来自同一位置

    \item Decoder Self-Attention: 防止看向后续位置（Masking）

    \item Encoder-Decoder Attention: Q来自Decoder，K、V来自Encoder

    \item Decoder采用Masked Attention确保自回归特性

    \item 实现了输入输出序列的全局依赖建模

\end{itemize}

\begin{frame}{位置编码与前馈网络}


\begin{itemize}

    \item 使用正弦/余弦函数生成位置编码

    \item PE(pos,2i) = sin(pos/100002i/dmodel)

    \item PE(pos,2i+1) = cos(pos/100002i/dmodel)

    \item 前馈网络：FFN(x) = max(0, xW1+b1)W2+b2

    \item 位置编码维度与词嵌入相同，可直接相加

\end{itemize}


\end{frame}


\begin{frame}{实验设置}


\begin{itemize}

    \item WMT 2014 English-German: 约450万句子对

    \item WMT 2014 English-French: 3600万句子对

    \item 使用Byte-pair Encoding (BPE) 进行分词

    \item 训练硬件：8块 NVIDIA P100 GPU

    \item 基础模型训练12小时，大模型训练3.5天

\end{itemize}


\end{frame}


\begin{frame}{机器翻译实验结果}


\begin{itemize}

    \item English-German: 28.4 BLEU (state-of-the-art)

    \item English-French: 41.8 BLEU (state-of-the-art)

    \item 比此前最佳模型提升超过2 BLEU

    \item 训练成本仅为最佳模型的1/4

    \item 基础模型也超越所有此前发布的单模型

\end{itemize}


\end{frame}


\begin{frame}{模型变种与消融分析}


\begin{itemize}

    \item 多头注意力中，头数过多或过少都会降低性能

    \item 减小attention key维度会损害模型质量

    \item 更大的模型和适当的dropout有助于性能提升

    \item 位置编码：正弦编码与学习编码效果相当

    \item 模型深度增加显著提升BLEU分数

\end{itemize}


\end{frame}
